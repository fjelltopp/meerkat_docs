    For a mobile clinic it finds the gps location for that daily register
    """
    

    ret=None
    mobile=False
    for dl in daily_loc.keys():
        if is_child(loc,dl,locations):
        #Need to deal with mobile ukraine location. 
            if locations[loc]["identifier"]=="deviceid" and interval==1:
                res=db.query_dict("SELECT * from %s where today=%s and deviceid=%s and gps is not null",(AsIs(daily_loc[dl]),start_time,locations[loc]["identifier_value"]))
                if res:
                    geo=res[0]["gps"].split()

                        ret= geo[0]+","+geo[1]
                    else:
                        ret=locations[loc]["geolocation"]
                else:
                    if locations[loc]["geolocation"]:
                        ret=locations[loc]["geolocation"]
            mobile=True
            break
    if not mobile:
        ret=locations[loc]["geolocation"]

    return ret

    """aggregates for yesterday"""
    #yesterday
    variables=get_variables(db)
    start_time=datetime.datetime.today()-datetime.timedelta(days=1)
    start_time=start_time.replace(hour=0,second=0,minute=0,microsecond=0)
    aggregate(start_time,1,variables,db)

    #YEA
    year=start_time.replace(month=1,day=1)
    aggregate(year,365,variables,db)

    #EPI Week - weeks since 1/1/XXXX
    days_since_jan=(start_time-year).days
    #epi_week=int(days_since_jan/7.0)+1
    start_epi_week=start_time-datetime.timedelta(days=days_since_jan%7.0)
